drop table if exists METADATA;

         ID varchar(20),
         VALUE varchar(20)
 );

drop table if exists BOOKSINVENTORY;
CREATE TABLE BOOKSINVENTORY (
	ID INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	BOX_ID varchar(20),
	BIBREC_SYS_NUM varchar(20) UNIQUE,
	ITEM_barcode varchar(20),
	BIBREC_CALL_NUM varchar(20),
    languageTerm varchar(30) NOT NULL,
    BIBREC_008_7_10 varchar(100),
	BIBREC_100a varchar(100),
	BIBREC_245a varchar(400),
	BIBREC_260b varchar(40),
	CHECKED INTEGER
);

         ID_TESTSET INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
         BATCHNAME VARCHAR(50) NOT NULL,
         SUBBATCHNAME VARCHAR(50) NOT NULL,

drop table if exists METS ;
create table METS (
  ID_METS INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
  ID_TESTSET INTEGER ,
  PATH varchar(50),
